📋 Apps Script 코드 보기 (배포용) 
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById('1yYkX0oA4v3YVQ2faECMOUJ60Egq6ETHtIV87kDscJls');
    var sheet = ss.getSheetByName('타임캡슐') || ss.insertSheet('타임캡슐');
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(['이름','학번','이메일','편지내용','발송예정일','발송완료']);
    }
    var sendDate = new Date();
    sendDate.setFullYear(sendDate.getFullYear() + 3);
    sheet.appendRow([data.name, data.studentId, data.email, data.letter,
      Utilities.formatDate(sendDate, 'Asia/Seoul', 'yyyy-MM-dd'), '미발송']);
    return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function sendTimeCapsuleEmails() {
  var ss = SpreadsheetApp.openById('1yYkX0oA4v3YVQ2faECMOUJ60Egq6ETHtIV87kDscJls');
  var sheet = ss.getSheetByName('타임캡슐');
  if (!sheet) return;
  var today = Utilities.formatDate(new Date(), 'Asia/Seoul', 'yyyy-MM-dd');
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    var sendDate = Utilities.formatDate(new Date(data[i][4]), 'Asia/Seoul', 'yyyy-MM-dd');
    if (sendDate <= today && data[i][5] === '미발송') {
      MailApp.sendEmail({
        to: data[i][2],
        subject: '📬 3년 전 네가 너에게 보낸 편지가 도착했어',
        body: '안녕, ' + data[i][0] + '!\n\n3년 전 너에게서 편지가 왔어.\n\n' + data[i][3]
      });
      sheet.getRange(i + 1, 6).setValue('발송완료 ' + today);
    }
  }
}
